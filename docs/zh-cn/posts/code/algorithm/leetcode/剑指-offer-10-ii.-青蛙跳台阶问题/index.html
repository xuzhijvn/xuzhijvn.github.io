剑指 Offer 10- II. 青蛙跳台阶问题

2021-10-11 1 分钟阅读 
class Solution {
    public int numWays(int n) {
        if(n < 2){
            return 1;
        }
        if(n == 2){
            return n;
        }
        int a = 1, b = 2, sum = 2;
        for(int i = 2; i < n; i++){
            sum = (a + b) % 1000000007;
            a = b;
            b = sum;
        }
        return sum;
    }
}
