剑指 Offer 55 - I. 二叉树的深度

Oct 11, 2021 1 min read 
class Solution {
    public int maxDepth(TreeNode root) {
        if(root == null){
            return 0;
        }
        if (root.left == null && root.right == null) {
            return 1;
        }
        
        int left = maxDepth(root.left);
        int right = maxDepth(root.right);
        return left > right ? left + 1 : right + 1;
    }
}
